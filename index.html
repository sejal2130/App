from flask import Flask, request, render_template_string import sqlite3 app = Flask(__name__) # ---------------- DATABASE ---------------- def get_db(): conn = sqlite3.connect("feedback.db") conn.row_factory = sqlite3.Row return conn conn = get_db() conn.execute(""" CREATE TABLE IF NOT EXISTS feedback ( id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT, message TEXT ) """) conn.commit() conn.close() # ---------------- HTML + CSS ---------------- HTML = """ Student Feedback

Student Feedback Form

All Feedback

{% for fb in feedbacks %}

{{ fb.name }}

{{ fb.message }}

{% endfor %} """ # ---------------- ROUTES ---------------- @app.route("/", methods=["GET", "POST"]) def index(): if request.method == "POST": name = request.form["name"] message = request.form["message"] conn = get_db() conn.execute("INSERT INTO feedback (name, message) VALUES (?, ?)", (name, message)) conn.commit() conn.close() conn = get_db() feedbacks = conn.execute("SELECT * FROM feedback").fetchall() conn.close() return render_template_string(HTML, feedbacks=feedbacks) # ---------------- RUN ---------------- if __name__ == "__main__": app.run(debug=True)
